Research: How to Build 2D in Three.js
Three.js 2D Approach (Best Practices):
Three.js is 3D but perfect for 2D because:

Option 1: Orthographic Camera (RECOMMENDED)
├── Use THREE.OrthographicCamera instead of PerspectiveCamera
├── Lock Z-axis movement (depth = layer order)
├── All objects are flat planes (PlaneGeometry)
├── Use Z position for draw order (sprites stacked)
├── Physics runs in 2D (X/Y only)
├── Perfect for platformers, top-down, puzzle games
└── Full Three.js features available (shaders, post-fx)

Option 2: Pure CSS/Canvas 2D (NOT recommended)
└── Would require rewriting everything, loses engine features

Option 3: Three.js Sprites
├── THREE.Sprite for billboard sprites
└── Less control than PlaneGeometry approach

BEST APPROACH: OrthographicCamera + PlaneGeometry
├── Sprites = PlaneGeometry + MeshBasicMaterial (texture)
├── Layers = Z-position (0.001 increments)
├── Physics = 2D wrapper (ignore Z axis)
├── Animation = Sprite sheet UV animation
└── Camera = OrthographicCamera with 2D controls
Coordinate System for 2D:
3D Engine:          2D Mode:
X = Right           X = Right (same)
Y = Up              Y = Up (same)
Z = Forward/Back    Z = Draw Layer (0-100)

2D World Space:
├── X: -∞ to +∞ (horizontal)
├── Y: -∞ to +∞ (vertical)
└── Z: 0.0 to 100.0 (layer depth)
    ├── 0.0 = Background
    ├── 50.0 = Midground
    └── 100.0 = Foreground

2D Camera System
Orthographic Camera Setup:
javascriptclass Camera2D {
  constructor(renderer, settings = {}) {
    this.renderer = renderer;
    this.settings = {
      pixelsPerUnit: settings.pixelsPerUnit || 100,
      zoom: settings.zoom || 1.0,
      position: settings.position || { x: 0, y: 0 },
      ...settings
    };

    this.setupCamera();
    this.setupControls();
  }

  setupCamera() {
    const aspect = window.innerWidth / window.innerHeight;
    const viewHeight = window.innerHeight / this.settings.pixelsPerUnit;
    const viewWidth = viewHeight * aspect;

    this.camera = new THREE.OrthographicCamera(
      -viewWidth / 2 * this.settings.zoom,   // left
       viewWidth / 2 * this.settings.zoom,   // right
       viewHeight / 2 * this.settings.zoom,  // top
      -viewHeight / 2 * this.settings.zoom,  // bottom
      -1000,                                  // near
       1000                                   // far
    );

    this.camera.position.set(
      this.settings.position.x,
      this.settings.position.y,
      100  // Z offset to see all layers
    );

    this.camera.lookAt(
      this.settings.position.x,
      this.settings.position.y,
      0
    );
  }

  setupControls() {
    // 2D viewport navigation
    this.isPanning = false;
    this.lastMousePos = new THREE.Vector2();

    const canvas = this.renderer.domElement;

    // Middle mouse pan
    canvas.addEventListener('mousedown', (e) => {
      if (e.button === 1) {
        this.isPanning = true;
        this.lastMousePos.set(e.clientX, e.clientY);
        e.preventDefault();
      }
    });

    canvas.addEventListener('mouseup', () => {
      this.isPanning = false;
    });

    canvas.addEventListener('mousemove', (e) => {
      if (!this.isPanning) return;

      const dx = (e.clientX - this.lastMousePos.x) / this.settings.pixelsPerUnit;
      const dy = (e.clientY - this.lastMousePos.y) / this.settings.pixelsPerUnit;

      this.camera.position.x -= dx * this.settings.zoom;
      this.camera.position.y += dy * this.settings.zoom;

      this.lastMousePos.set(e.clientX, e.clientY);
    });

    // Scroll to zoom
    canvas.addEventListener('wheel', (e) => {
      e.preventDefault();
      const zoomFactor = e.deltaY > 0 ? 1.1 : 0.9;
      this.setZoom(this.settings.zoom * zoomFactor);
    });

    // Keyboard pan (Arrow keys)
    window.addEventListener('keydown', (e) => {
      if (document.activeElement.tagName === 'INPUT') return;
      const speed = 5 * this.settings.zoom;
      switch (e.key) {
        case 'ArrowLeft':  this.camera.position.x -= speed; break;
        case 'ArrowRight': this.camera.position.x += speed; break;
        case 'ArrowUp':    this.camera.position.y += speed; break;
        case 'ArrowDown':  this.camera.position.y -= speed; break;
        case 'f': case 'F': this.focusOnSelection(); break;
      }
    });
  }

  setZoom(zoom) {
    this.settings.zoom = Math.max(0.1, Math.min(20, zoom));
    const aspect = window.innerWidth / window.innerHeight;
    const viewHeight = window.innerHeight / this.settings.pixelsPerUnit;
    const viewWidth = viewHeight * aspect;

    this.camera.left   = -viewWidth  / 2 * this.settings.zoom;
    this.camera.right  =  viewWidth  / 2 * this.settings.zoom;
    this.camera.top    =  viewHeight / 2 * this.settings.zoom;
    this.camera.bottom = -viewHeight / 2 * this.settings.zoom;

    this.camera.updateProjectionMatrix();
  }

  worldToScreen(worldPos) {
    const vector = new THREE.Vector3(worldPos.x, worldPos.y, 0);
    vector.project(this.camera);

    return {
      x: (vector.x + 1) / 2 * window.innerWidth,
      y: (-vector.y + 1) / 2 * window.innerHeight
    };
  }

  screenToWorld(screenPos) {
    const vector = new THREE.Vector3(
      (screenPos.x / window.innerWidth) * 2 - 1,
      -((screenPos.y / window.innerHeight) * 2 - 1),
      0
    );
    vector.unproject(this.camera);

    return { x: vector.x, y: vector.y };
  }

  focusOnSelection(objects) {
    if (!objects || objects.length === 0) return;

    const box = new THREE.Box3();
    objects.forEach(obj => box.expandByObject(obj));

    const center = box.getCenter(new THREE.Vector3());
    const size = box.getSize(new THREE.Vector3());

    this.camera.position.x = center.x;
    this.camera.position.y = center.y;

    const requiredZoom = Math.max(
      size.x / (window.innerWidth / this.settings.pixelsPerUnit),
      size.y / (window.innerHeight / this.settings.pixelsPerUnit)
    ) * 1.5;

    this.setZoom(requiredZoom);
  }

  follow(target, smoothing = 0.1) {
    // Camera follow for runtime
    this.followTarget = target;
    this.followSmoothing = smoothing;
  }

  update(deltaTime) {
    if (this.followTarget) {
      const targetX = this.followTarget.position.x;
      const targetY = this.followTarget.position.y;

      this.camera.position.x += (targetX - this.camera.position.x) * this.followSmoothing;
      this.camera.position.y += (targetY - this.camera.position.y) * this.followSmoothing;
    }
  }
}

SpriteRenderer Component
javascriptclass SpriteRenderer {
  constructor(actor, properties = {}) {
    this.actor = actor;
    this.type = 'SpriteRenderer';

    this.properties = {
      texture: null,
      color: '#FFFFFF',
      flipX: false,
      flipY: false,
      sortingLayer: 'Default',
      orderInLayer: 0,
      pixelsPerUnit: 100,
      pivot: { x: 0.5, y: 0.5 },
      ...properties
    };

    this.mesh = null;
    this.material = null;

    this.create();
  }

  create() {
    const ppu = this.properties.pixelsPerUnit;

    let width = 1;
    let height = 1;

    if (this.properties.texture) {
      const tex = this.properties.texture;
      width = tex.image.width / ppu;
      height = tex.image.height / ppu;
    }

    // Create plane geometry (the sprite quad)
    const geometry = new THREE.PlaneGeometry(width, height);

    // Adjust pivot point
    const pivotOffsetX = (0.5 - this.properties.pivot.x) * width;
    const pivotOffsetY = (0.5 - this.properties.pivot.y) * height;
    geometry.translate(pivotOffsetX, pivotOffsetY, 0);

    // Sprite material
    this.material = new THREE.MeshBasicMaterial({
      map: this.properties.texture,
      color: new THREE.Color(this.properties.color),
      transparent: true,
      alphaTest: 0.01,
      side: THREE.DoubleSide
    });

    this.mesh = new THREE.Mesh(geometry, this.material);
    this.mesh.userData.componentType = 'SpriteRenderer';
    this.mesh.userData.isActor = true;
    this.mesh.userData.actorId = this.actor.actorId;

    // Apply flip
    if (this.properties.flipX) this.mesh.scale.x = -1;
    if (this.properties.flipY) this.mesh.scale.y = -1;

    // Set Z based on layer
    this.updateSortingOrder();

    this.actor.group.add(this.mesh);
  }

  updateSortingOrder() {
    // Use Z position for sorting
    // Higher Z = drawn on top
    const layerZ = this.getSortingLayerZ(this.properties.sortingLayer);
    const orderZ = this.properties.orderInLayer * 0.001;
    this.mesh.position.z = layerZ + orderZ;
  }

  getSortingLayerZ(layerName) {
    const layers = {
      'Background': 0,
      'BackgroundDetail': 1,
      'Ground': 10,
      'Default': 20,
      'Characters': 30,
      'Projectiles': 40,
      'Effects': 50,
      'UI': 90,
      'Overlay': 100
    };
    return layers[layerName] ?? 20;
  }

  setSprite(texture) {
    this.properties.texture = texture;
    this.material.map = texture;
    this.material.needsUpdate = true;

    // Update size from texture
    const ppu = this.properties.pixelsPerUnit;
    const width = texture.image.width / ppu;
    const height = texture.image.height / ppu;

    this.mesh.geometry.dispose();
    this.mesh.geometry = new THREE.PlaneGeometry(width, height);
  }

  setColor(color) {
    this.properties.color = color;
    this.material.color.set(color);
  }

  setFlipX(flip) {
    this.properties.flipX = flip;
    this.mesh.scale.x = flip ? -1 : 1;
  }

  setOpacity(opacity) {
    this.material.opacity = opacity;
  }

  setUVRect(x, y, width, height) {
    // For sprite sheet animation
    const uv = this.mesh.geometry.attributes.uv;

    uv.setXY(0, x, y + height);
    uv.setXY(1, x + width, y + height);
    uv.setXY(2, x, y);
    uv.setXY(3, x + width, y);

    uv.needsUpdate = true;
  }
}
RigidBody2D Component
javascriptclass RigidBody2D {
  constructor(actor, properties = {}) {
    this.actor = actor;
    this.type = 'RigidBody2D';

    this.properties = {
      bodyType: 'Dynamic', // Dynamic, Static, Kinematic
      mass: 1.0,
      gravityScale: 1.0,
      linearDrag: 0,
      angularDrag: 0.05,
      freezeRotation: true,
      ...properties
    };

    // Physics state (pure 2D)
    this.velocity = new THREE.Vector2(0, 0);
    this.acceleration = new THREE.Vector2(0, 0);
    this.force = new THREE.Vector2(0, 0);
    this.angularVelocity = 0;

    this.isGrounded = false;
    this.isTouching = {
      left: false,
      right: false,
      top: false,
      bottom: false
    };
  }

  addForce(x, y) {
    if (this.properties.bodyType !== 'Dynamic') return;
    this.force.x += x;
    this.force.y += y;
  }

  addImpulse(x, y) {
    if (this.properties.bodyType !== 'Dynamic') return;
    this.velocity.x += x / this.properties.mass;
    this.velocity.y += y / this.properties.mass;
  }

  setVelocity(x, y) {
    this.velocity.set(x, y);
  }

  setVelocityX(x) {
    this.velocity.x = x;
  }

  setVelocityY(y) {
    this.velocity.y = y;
  }

  update(deltaTime, gravity) {
    if (this.properties.bodyType !== 'Dynamic') return;

    // Apply gravity
    this.force.y += gravity * this.properties.gravityScale * this.properties.mass;

    // Apply forces to velocity
    this.acceleration.x = this.force.x / this.properties.mass;
    this.acceleration.y = this.force.y / this.properties.mass;

    this.velocity.x += this.acceleration.x * deltaTime;
    this.velocity.y += this.acceleration.y * deltaTime;

    // Apply drag
    this.velocity.x *= Math.max(0, 1 - this.properties.linearDrag * deltaTime);
    this.velocity.y *= Math.max(0, 1 - this.properties.linearDrag * deltaTime);

    // Move actor
    this.actor.group.position.x += this.velocity.x * deltaTime;
    this.actor.group.position.y += this.velocity.y * deltaTime;

    // Reset forces
    this.force.set(0, 0);
  }
}

javascriptclass BoxCollider2D {
  constructor(actor, properties = {}) {
    this.actor = actor;
    this.type = 'BoxCollider2D';

    this.properties = {
      size: { x: 1, y: 1 },
      offset: { x: 0, y: 0 },
      isTrigger: false,
      layer: 'Default',
      ...properties
    };

    this.debugMesh = this.createDebugMesh();
  }

  createDebugMesh() {
    const geo = new THREE.EdgesGeometry(
      new THREE.BoxGeometry(
        this.properties.size.x,
        this.properties.size.y,
        0.01
      )
    );
    const mat = new THREE.LineBasicMaterial({
      color: this.properties.isTrigger ? 0x00FFFF : 0x00FF00,
      transparent: true,
      opacity: 0.8,
      depthTest: false
    });

    const mesh = new THREE.LineSegments(geo, mat);
    mesh.position.x = this.properties.offset.x;
    mesh.position.y = this.properties.offset.y;
    mesh.position.z = 200;
    mesh.visible = false;
    mesh.userData.isDebugGizmo = true;

    this.actor.group.add(mesh);
    return mesh;
  }

  getBounds() {
    const worldPos = new THREE.Vector3();
    this.actor.group.getWorldPosition(worldPos);

    const halfW = this.properties.size.x / 2;
    const halfH = this.properties.size.y / 2;
    const offsetX = this.properties.offset.x;
    const offsetY = this.properties.offset.y;

    return {
      left:   worldPos.x + offsetX - halfW,
      right:  worldPos.x + offsetX + halfW,
      top:    worldPos.y + offsetY + halfH,
      bottom: worldPos.y + offsetY - halfH,
      centerX: worldPos.x + offsetX,
      centerY: worldPos.y + offsetY
    };
  }

  intersects(other) {
    const a = this.getBounds();
    const b = other.getBounds();

    return a.left < b.right &&
           a.right > b.left &&
           a.bottom < b.top &&
           a.top > b.bottom;
  }

  getOverlap(other) {
    const a = this.getBounds();
    const b = other.getBounds();

    const overlapX = Math.min(a.right, b.right) - Math.max(a.left, b.left);
    const overlapY = Math.min(a.top, b.top) - Math.max(a.bottom, b.bottom);

    return { x: overlapX, y: overlapY };
  }

  setDebugVisible(visible) {
    this.debugMesh.visible = visible;
  }
}

2D Animation Instance (Reuses Architecture):
javascriptclass AnimationInstance2D {
  constructor(animationBlueprint, spriteRenderer) {
    this.blueprint = animationBlueprint;
    this.spriteRenderer = spriteRenderer;
    this.spriteSheet = null;

    this.variables = {};
    this.currentState = null;
    this.currentFrame = 0;
    this.frameTimer = 0;
    this.currentAnimation = null;

    this.stateMachine = null;

    this.initialize();
  }

  initialize() {
    // Load sprite sheet
    this.spriteSheet = this.loadSpriteSheet(
      this.blueprint.spriteSheet
    );

    // Initialize variables (SAME AS 3D!)
    this.blueprint.variables.forEach(v => {
      this.variables[v.name] = v.default;
    });

    // Setup state machine (SAME ARCHITECTURE AS 3D!)
    this.stateMachine = this.blueprint.animationGraph.stateMachines[0];
    this.currentState = this.stateMachine.entryState;
  }

  update(deltaTime, owner) {
    // Execute Event Graph (SAME AS 3D!)
    this.executeEventGraph(owner);

    // Evaluate State Machine (SAME AS 3D!)
    this.evaluateStateMachine();

    // Update current animation frame
    this.updateFrame(deltaTime);
  }

  executeEventGraph(owner) {
    // Get speed from owner
    if (owner?.rigidBody2D) {
      const vel = owner.rigidBody2D.velocity;
      this.variables.speed = Math.abs(vel.x);
      this.variables.isGrounded = owner.rigidBody2D.isGrounded;
      this.variables.isJumping = owner.rigidBody2D.velocity.y > 0 && !owner.rigidBody2D.isGrounded;
      this.variables.isFalling = owner.rigidBody2D.velocity.y < -10 && !owner.rigidBody2D.isGrounded;
    }
  }

  evaluateStateMachine() {
    // EXACT SAME LOGIC AS 3D STATE MACHINE!
    const sm = this.stateMachine;

    for (const transition of sm.transitions) {
      if (transition.from === this.currentState || transition.from === '*') {
        if (this.evaluateCondition(transition.condition)) {
          if (transition.to !== this.currentState) {
            this.transitionToState(transition.to);
            break;
          }
        }
      }
    }
  }

  evaluateCondition(condition) {
    // SAME CONDITION EVALUATION AS 3D!
    const vars = this.variables;
    try {
      return new Function(...Object.keys(vars),
        `return ${condition}`
      )(...Object.values(vars));
    } catch (e) {
      return false;
    }
  }

  transitionToState(newStateId) {
    this.currentState = newStateId;

    const state = this.stateMachine.states.find(
      s => s.stateId === newStateId
    );

    if (state) {
      // Find animation in sprite sheet
      this.currentAnimation = this.spriteSheet.animations.find(
        a => a.animId === state.animation2D
      );
      this.currentFrame = 0;
      this.frameTimer = 0;
    }
  }

  updateFrame(deltaTime) {
    if (!this.currentAnimation) return;

    const fps = this.currentAnimation.fps;
    this.frameTimer += deltaTime;

    if (this.frameTimer >= 1.0 / fps) {
      this.frameTimer -= 1.0 / fps;

      if (this.currentAnimation.loop) {
        this.currentFrame = (this.currentFrame + 1) %
          this.currentAnimation.frames.length;
      } else {
        this.currentFrame = Math.min(
          this.currentFrame + 1,
          this.currentAnimation.frames.length - 1
        );
      }

      this.applyFrame();
    }
  }

  applyFrame() {
    if (!this.currentAnimation || !this.spriteSheet) return;

    const frameId = this.currentAnimation.frames[this.currentFrame];
    const sprite = this.spriteSheet.sprites.find(s => s.spriteId === frameId);

    if (sprite) {
      // Update UV coordinates on the sprite mesh
      const texW = this.spriteSheet.textureWidth;
      const texH = this.spriteSheet.textureHeight;

      const u = sprite.x / texW;
      const v = 1 - (sprite.y + sprite.height) / texH;
      const uw = sprite.width / texW;
      const uh = sprite.height / texH;

      this.spriteRenderer.setUVRect(u, v, uw, uh);
    }
  }
}

javascriptclass Physics2DWorld {
  constructor(settings = {}) {
    this.gravity = settings.gravity || { x: 0, y: -980 };
    this.bodies = [];
    this.colliders = [];
    this.contacts = [];

    this.pixelsPerUnit = settings.pixelsPerUnit || 100;
  }

  addBody(rigidBody2D) {
    this.bodies.push(rigidBody2D);
  }

  addCollider(collider2D) {
    this.colliders.push(collider2D);
  }

  step(deltaTime) {
    // Fixed timestep for stable physics
    const fixedDT = Math.min(deltaTime, 0.033);

    // Update all dynamic bodies
    this.bodies.forEach(body => {
      if (body.properties.bodyType === 'Dynamic') {
        body.update(fixedDT, this.gravity.y);
      }
    });

    // Collision detection and resolution
    this.detectAndResolveCollisions();

    // Clear grounded flags
    this.bodies.forEach(body => {
      body.isGrounded = false;
      body.isTouching.left = false;
      body.isTouching.right = false;
      body.isTouching.top = false;
      body.isTouching.bottom = false;
    });

    // Update grounded from contacts
    this.contacts.forEach(contact => {
      this.resolveContact(contact);
    });
  }

  detectAndResolveCollisions() {
    this.contacts = [];

    for (let i = 0; i < this.colliders.length; i++) {
      for (let j = i + 1; j < this.colliders.length; j++) {
        const a = this.colliders[i];
        const b = this.colliders[j];

        if (a.intersects(b)) {
          const overlap = a.getOverlap(b);
          this.contacts.push({ a, b, overlap });

          // Fire overlap events if trigger
          if (a.properties.isTrigger || b.properties.isTrigger) {
            this.fireOverlapEvent(a, b);
          } else {
            this.resolvePhysicsCollision(a, b, overlap);
          }
        }
      }
    }
  }

  resolvePhysicsCollision(a, b, overlap) {
    const bodyA = a.actor.getRigidBody2D();
    const bodyB = b.actor.getRigidBody2D();

    if (!bodyA && !bodyB) return;

    // Find minimum overlap axis
    if (Math.abs(overlap.x) < Math.abs(overlap.y)) {
      // Horizontal collision
      const dir = a.getBounds().centerX < b.getBounds().centerX ? -1 : 1;

      if (bodyA && bodyA.properties.bodyType === 'Dynamic') {
        a.actor.group.position.x += overlap.x * dir * 0.5;
        bodyA.velocity.x = 0;
        bodyA.isTouching.left = dir > 0;
        bodyA.isTouching.right = dir < 0;
      }
      if (bodyB && bodyB.properties.bodyType === 'Dynamic') {
        b.actor.group.position.x -= overlap.x * dir * 0.5;
        bodyB.velocity.x = 0;
        bodyB.isTouching.left = dir < 0;
        bodyB.isTouching.right = dir > 0;
      }

    } else {
      // Vertical collision
      const dir = a.getBounds().centerY < b.getBounds().centerY ? -1 : 1;

      if (bodyA && bodyA.properties.bodyType === 'Dynamic') {
        a.actor.group.position.y += overlap.y * dir * 0.5;

        if (dir > 0) {
          // Landing on top
          bodyA.velocity.y = Math.max(0, bodyA.velocity.y);
          bodyA.isGrounded = true;
          bodyA.isTouching.bottom = true;
        } else {
          bodyA.velocity.y = Math.min(0, bodyA.velocity.y);
          bodyA.isTouching.top = true;
        }
      }

      if (bodyB && bodyB.properties.bodyType === 'Dynamic') {
        b.actor.group.position.y -= overlap.y * dir * 0.5;

        if (dir < 0) {
          bodyB.velocity.y = Math.max(0, bodyB.velocity.y);
          bodyB.isGrounded = true;
          bodyB.isTouching.bottom = true;
        } else {
          bodyB.velocity.y = Math.min(0, bodyB.velocity.y);
          bodyB.isTouching.top = true;
        }
      }
    }
  }

  fireOverlapEvent(a, b) {
    a.actor.emit('onBeginOverlap', { other: b.actor });
    b.actor.emit('onBeginOverlap', { other: a.actor });
  }
}

javascriptclass TilemapRenderer {
  constructor(scene, tilemapData) {
    this.scene = scene;
    this.data = tilemapData;
    this.group = new THREE.Group();
    this.group.name = tilemapData.assetName;

    this.collisionBodies = [];

    this.build();
    scene.add(this.group);
  }

  build() {
    this.data.layers.forEach(layer => {
      if (layer.visible) {
        this.buildLayer(layer);
      }
    });
  }

  buildLayer(layer) {
    const layerGroup = new THREE.Group();
    layerGroup.name = layer.layerName;

    const tileSize = this.data.tileSize;
    const ppu = this.data.pixelsPerUnit;
    const tileSizeWorld = tileSize.x / ppu;

    Object.entries(layer.tiles).forEach(([coordStr, tileId]) => {
      const [x, y] = coordStr.split(',').map(Number);
      const tileData = this.getTileData(tileId);

      if (!tileData) return;

      const sprite = this.createTileSprite(tileData, tileSizeWorld);
      sprite.position.x = x * tileSizeWorld + tileSizeWorld / 2;
      sprite.position.y = y * tileSizeWorld + tileSizeWorld / 2;
      sprite.position.z = layer.sortingOrder * 0.01;

      layerGroup.add(sprite);

      // Generate collision for solid tiles
      if (layer.hasCollision && tileData.collision) {
        this.generateTileCollision(x, y, tileData, tileSizeWorld);
      }
    });

    this.group.add(layerGroup);
  }

  createTileSprite(tileData, size) {
    const geo = new THREE.PlaneGeometry(size, size);
    const mat = new THREE.MeshBasicMaterial({
      map: this.getTileTexture(tileData),
      transparent: true,
      alphaTest: 0.01
    });
    return new THREE.Mesh(geo, mat);
  }

  generateTileCollision(x, y, tileData, tileSize) {
    const bounds = {
      left:   x * tileSize,
      right:  x * tileSize + tileSize,
      top:    y * tileSize + tileSize,
      bottom: y * tileSize,
      centerX: x * tileSize + tileSize / 2,
      centerY: y * tileSize + tileSize / 2
    };

    this.collisionBodies.push({
      type: 'static',
      collisionType: tileData.collision,
      bounds: bounds
    });
  }
}
